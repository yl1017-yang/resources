틀린그림 찾기능력이 향상되는 Destructuring 문법

        
          var arr = [2,3,4];

          var a = arr[0];
          var b = arr[1];

          var [a,b,c] = [2,3,4]; //array 데이터를 전부 변수에 담으러면
          var [a,b,c = 10] = [2,3];  //array destructuring 할때 몇개를 빼먹는 다면 - 등호로 기본값 지정 가능

          
          var obj = { name : 'kim', age : 30 };
          var name = obj.name;
          var age = obj.age;

          var { name, age } = { name : 'kim', age : 30 };
          var { name, age = 31 } = { name : 'kim' };
          var { name : 나이 } = { name : 'kim' };
          var { name : 나이 = 'park' } = {  };



          var name = 'kim';
          var age = 30;

          var obj = { name : name, age : age };
          var obj = { name, age };  //es6 문법


          함수 파라미터 만들때도 destructuring 문법 사용가능
