          backquote, backtick 문자열의 장점 : 
          1. 엔터키를 쳐도 문자 인식이 됨
          2. 중간에 변수넣기 편함          

          var 변수 = '손흥민';
          var 문자 = '안녕하세요 저는' + 변수 + '저는' ;
          console.log(문자);

          var 문자 = `안녕하세요 저는 ${ 변수 }`;
          var 템플릿 = `
 ${ 변수 } 
`;  //문자열 사용시


          ### tagged literal(태그드 리터럴) 사용법 : `문자`를 해체분석할 수 있음
          var 변수 = '손흥민';
          var 문자 = `안녕하세요 ${ 변수 } 입니다`;

          function 함수() {
            return 10
          }
          console.log(함수`안녕하세요 ${ 변수 } 입니다`)  //함수() 대신 사용

          function 해체분석기(문자들, 변수들) {
            console.log(문자들);
            console.log(변수들);

            console.log( 문자들[1] + 변수들 + 문자들[0] );
          }
          해체분석기`안녕하세요 ${ 변수 } 입니다`

          //순서변경하고 싶을때 : 쌩자바스크립트는 순서 바꿀려면 찾아서 해야함
          console.log( 문자들[1] + 변수들 + 문자들[0] );

첫번째 예시 : 제작 후 양말이 20개였음

          `바지${pants} 양말${socks}`;

          제작 후 양말이 20개였음
          function 해체분석기(문자들, 변수들1, 변수들2) {
            console.log( `문자들[1] + 변수들1 + 문자들[0] + 변수들2` );
          }
          해체분석기`바지${pants} 양말${socks}`;

          ※ 변수가 여러개일 경우 ...spread operator 사용 [...변수들]

두번째 예시 : pants가 0개면 '바지안팔아요 양말100' 이라는 문자로 바꾸는 해체분석기 제작

        
          var pants = 0;
          var socks = 100;
          `바지${pants} 양말${socks}`;

          function 해체분석기(글자들, 변수들1, 변수들2) {
            if (변수들1 == 0 ) {
              console.log(`바지다팔렸어요 + 양말` + 변수들2);
            }
          }
          해체분석기`바지${pants} 양말${socks}`;

Spread Operator -- array를 파라미터형태로 집어넣고 싶을 때 씁니다. 

        
          function 더하기(a,b,c){
            console.log(a + b + c)
         }

         var 어레이 = [10, 20, 30];

         더하기(...어레이);  //요즘방식
         더하기.apply(undefined, 어레이);  //옛날방식(몰라도됨)
